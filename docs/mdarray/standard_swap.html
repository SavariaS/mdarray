std::swap(mdlib::mdarray)

  template<typename T, std::size_t... N>
  void swap(mdlib::mdarray<T, N...>& lhs, mdlib::mdarray<T, N...>& rhs);

Specializes std::swap. Swaps the contents of the two containers.


Parameters

lhs, rhs - containers to swap.


Return value

(none)


Example

#include <mdarray.hpp>
#include <iostream>

int main()
{
    mdlib::mdarray<int, 2, 2> arr1 = {1, 0,
                                      0, 1};

    mdlib::mdarray<int, 2, 2> arr2 = {1, 2,
                                      3, 4};
    
    std::swap(arr1, arr2);

    std::cout << "arr1:\n";
    for(auto elem : arr1)
    {
        std::cout << elem << ' ';
    }
    std::cout << "\n\n";

    std::cout << "arr2:\n";
    for(auto elem : arr2)
    {
        std::cout << elem << ' ';
    }
    std::cout << '\n';
}

Output

arr1:
1 2 3 4 

arr2:
1 0 0 1
