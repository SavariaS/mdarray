xcontainer::mdarray<T, N...>::swap

  constexpr void swap(mdarray<T, N...>& other);

Exchange the contents with another container.


Parameters

other - the container to exchange contents with.


Return value

(none)


Example

#include <mdarray.hpp>
#include <iostream>

int main()
{
    xcontainer::mdarray<int, 2, 2> arr1 = {1, 0,
                                        0, 1};

    xcontainer::mdarray<int, 2, 2> arr2 = {1, 2,
                                        3, 4};
    
    arr1.swap(arr2);

    std::cout << "arr1:\n";
    for(auto elem : arr1)
    {
        std::cout << elem << ' ';
    }
    std::cout << "\n\n";

    std::cout << "arr2:\n";
    for(auto elem : arr2)
    {
        std::cout << elem << ' ';
    }
    std::cout << '\n';
}

Output

arr1:
1 2 3 4 

arr2:
1 0 0 1
