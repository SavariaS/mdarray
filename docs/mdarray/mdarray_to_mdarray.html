std::to_mdarray

  template<typename T, std::size_t N>
  constexpr mdlib::from_array<std::remove_cv_t<T[N]>> to_mdarray(T (&arr)[N])

Creates a mdarray from a built-in multidimensional array.


Parameters

arr - the built-in multidimensional array to be converted to mdarray.


Return value

The corresponding mdarray.


Example

#include <mdarray.hpp>
#include <iostream>

int main()
{
    int builtin_matrice[3][3] = {1, 0, 0,
                                 0, 1, 0,
                                 0, 0, 1};

    mdlib::mdarray<int, 3, 3> from_builtin = std::to_mdarray(builtin_matrice);

    std::cout << "Built-in:\n";
    for(int i = 0; i < 3; ++i)
    {
        for(int j = 0; j < 3; ++j)
        {
            std::cout << builtin_matrice[i][j] << ' ';
        }
        std::cout << '\n';
    }

    std::cout << "\nmdarray:\n";
    for(int i = 0; i < 3; ++i)
    {
        for(int j = 0; j < 3; ++j)
        {
            std::cout << from_builtin[i][j] << ' ';
        }
        std::cout << '\n';
    }
}

Output

Built-in:
1 0 0 
0 1 0 
0 0 1 

mdarray:
1 0 0 
0 1 0 
0 0 1
